let resources = []; const resourceForm = document.querySelector('#resource-form'); const resourcesTbody = document.querySelector('#resources-tbody'); function createResourceRow(resource) { const tr = document.createElement('tr'); const titleTd = document.createElement('td'); titleTd.textContent = resource.title; const descriptionTd = document.createElement('td'); descriptionTd.textContent = resource.description; const linkTd = document.createElement('td'); const linkAnchor = document.createElement('a'); linkAnchor.href = resource.link; linkAnchor.textContent = resource.link; linkAnchor.target = '_blank'; linkTd.appendChild(linkAnchor); const actionsTd = document.createElement('td'); const editBtn = document.createElement('button'); editBtn.textContent = 'Edit'; editBtn.addEventListener('click', () => editResource(resource.id)); const deleteBtn = document.createElement('button'); deleteBtn.textContent = 'Delete'; deleteBtn.addEventListener('click', () => deleteResource(resource.id)); actionsTd.appendChild(editBtn); actionsTd.appendChild(deleteBtn); tr.appendChild(titleTd); tr.appendChild(descriptionTd); tr.appendChild(linkTd); tr.appendChild(actionsTd); return tr; } function renderTable(resourceList = resources) { resourcesTbody.innerHTML = ''; resourceList.forEach(resource => { const row = createResourceRow(resource); resourcesTbody.appendChild(row); }); } async function fetchResources() { try { const response = await fetch('resources.php'); const result = await response.json(); if (result.success) { resources = result.data; renderTable(resources); } else { resources = []; renderTable(resources); } } catch (error) { console.error('Error fetching resources:', error); resources = []; renderTable(resources); } } async function addResource(resourceData) { try { const response = await fetch('resources.php', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(resourceData) }); const result = await response.json(); if (result.success) { fetchResources(); resourceForm.reset(); } else { alert(result.message || 'Failed to add resource'); } } catch (error) { console.error('Error adding resource:', error); } } async function deleteResource(id) { try { const response = await fetch(`resources.php?id=${id}`, { method: 'DELETE' }); const result = await response.json(); if (result.success) { fetchResources(); } else { alert(result.message || 'Failed to delete resource'); } } catch (error) { console.error('Error deleting resource:', error); } } function editResource(id) { const resource = resources.find(r => r.id === id); if (!resource) return; document.querySelector('#title').value = resource.title; document.querySelector('#description').value = resource.description; document.querySelector('#link').value = resource.link; resourceForm.dataset.editId = id; } async function updateResource(id, resourceData) { try { const response = await fetch('resources.php', { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id, ...resourceData }) }); const result = await response.json(); if (result.success) { delete resourceForm.dataset.editId; resourceForm.reset(); fetchResources(); } else { alert(result.message || 'Failed to update resource'); } } catch (error) { console.error('Error updating resource:', error); } } resourceForm.addEventListener('submit', function (e) { e.preventDefault(); const resourceData = { title: document.querySelector('#title').value.trim(), description: document.querySelector('#description').value.trim(), link: document.querySelector('#link').value.trim() }; const editId = resourceForm.dataset.editId; if (editId) { updateResource(Number(editId), resourceData); } else { addResource(resourceData); } }); fetchResources();
